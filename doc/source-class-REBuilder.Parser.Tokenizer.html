 112:     /**
 113:      * Constructor
 114:      *

 116:      * @param callable $receiver Function that will receive the emitted tokens
 117:      * 
 118:      * @throws \REBuilder\Exception\EmptyRegex
 119:      */
 120:     public function __construct ($regex, $receiver)
 121:     {
 122:         //Check if the regex is empty
 123:         if ($regex === "") {
 124:             throw new Exception\EmptyRegex();
 125:         }
 126: 
 127:         $this->_regex = $regex;
 128:         $this->_receiver = $receiver;
 129:         $this->_modifiersStack = new \SplStack;
 130:     }

 132:     /**
 133:      * Starts the tokenization proces
 134:      * 
 135:      * @return void
 136:      */
 137:     public function tokenize ()
 138:     {
 139:         //Since delimiters are the only exception to the normal regex syntax and
 140:         //the tokenizer needs to know regex modifiers to handle some situations,
 141:         //parse them immediately and strip them from the regex
 142:         list(
 143:             $delimiter, $endDelimiter, $rModifiers
 144:         ) = $this->_stripDelimitersAndModifiers();
 145:         $checkEndDelimiter = $delimiter === $endDelimiter;
 146: 
 147:         //Store regex length
 148:         $this->_length = strlen($this->_regex);
 149: 
 150:         //Loop regex characters
 151:         while (($char = $this->_consume()) !== null) {
 152:             //If character is backslash and it's not escaped
 153:             if ($char === "\\" && !$this->_escaped) {
 154:                 //Set escaped flag to true
 155:                 $this->_escaped  = true;
 156:                 continue;
 157:             }
 158:             //If not in char class and not escaped and it's a dot
 159:             elseif (!$this->_inCharClass && !$this->_escaped && $char === ".") {
 160:                 //Emit a dot token
 161:                 $this->_emitToken(
 162:                     Token::TYPE_DOT,
 163:                     $char
 164:                 );
 165:             }
 166:             //If escaped and it's a generic character type identifier
 167:             elseif ($this->_escaped &&
 168:                     Rules::validateGenericCharType($char)) {
 169:                 //Emit a generic character type token
 170:                 $this->_emitToken(
 171:                     Token::TYPE_GENERIC_CHAR_TYPE,
 172:                     $char
 173:                 );
 174:             }
 175:             //If not in char class and escaped and it's a simple assertion identifier
 176:             elseif (!$this->_inCharClass && $this->_escaped &&
 177:                     Rules::validateSimpleAssertion($char)) {
 178:                 //Emit a simple assertion token
 179:                 $this->_emitToken(
 180:                     Token::TYPE_SIMPLE_ASSERTION,
 181:                     $char
 182:                 );
 183:             }
 184:             //If escaped and it's a non-printing character identifier
 185:             elseif ($this->_escaped &&
 186:                     Rules::validateNonPrintingChar($char)) {
 187:                 //Emit a non-printing character token
 188:                 $this->_emitToken(
 189:                     Token::TYPE_NON_PRINTING_CHAR,
 190:                     $char
 191:                 );
 192:             }
 193:             //If not in char class and escaped and it's the extended unicode
 194:             //sequence identifier
 195:             elseif (!$this->_inCharClass && $this->_escaped && $char === "X") {
 196:                 //Emit a extended unicode sequence token
 197:                 $this->_emitToken(
 198:                     Token::TYPE_EXT_UNICODE_SEQUENCE,
 199:                     $char
 200:                 );
 201:             }
 202:             //If not in char class and escaped and it's a single byte identifier
 203:             elseif (!$this->_inCharClass && $this->_escaped && $char === "C") {
 204:                 //Emit a single byte identifier token
 205:                 $this->_emitToken(
 206:                     Token::TYPE_BYTE,
 207:                     $char
 208:                 );
 209:             }
 210:             //If not in char class and not escaped and it's an anchor
 211:             elseif (!$this->_inCharClass && !$this->_escaped &&
 212:                     ($char === "^" || $char === "$")) {
 213:                 //Emit an anchor token
 214:                 $this->_emitToken(
 215:                     $char === "^" ? 
 216:                         Token::TYPE_START_ANCHOR :
 217:                         Token::TYPE_END_ANCHOR,
 218:                     $char
 219:                 );
 220:             }
 221:             //If not in char class and not escaped and it's a pipe
 222:             elseif (!$this->_inCharClass && !$this->_escaped && $char === "|") {
 223:                 //Emit an alternation identifier token
 224:                 $this->_emitToken(
 225:                     Token::TYPE_ALTERNATION,
 226:                     $char
 227:                 );
 228:             }
 229:             //If not in char class and escaped and it's the unicode character
 230:             //class identifier
 231:             elseif (!$this->_inCharClass && $this->_escaped &&
 232:                     ($char === "p" || $char === "P")) {
 233:                 //Take the next character
 234:                 $nextChar = $this->_consume();
 235:                 //If there are no characters left throw an exception
 236:                 if ($nextChar === null) {
 237:                     throw new Exception\Generic(
 238:                         "Unspecified character class form \\" . $char
 239:                     );
 240:                 }
 241:                 //If the next char is not { emit the token
 242:                 elseif ($nextChar !== "{") {
 243:                     $this->_emitToken(
 244:                         Token::TYPE_UNICODE_CHAR_CLASS,
 245:                         $char,
 246:                         $nextChar
 247:                     );
 248:                 }
 249:                 //If the next char is {
 250:                 else {
 251:                     //Find everything until the closing bracket
 252:                     $nextChars = $this->_consumeUntil("}", true);
 253:                     //If the closing bracket has not been found throw an
 254:                     //exception
 255:                     if ($nextChars === null) {
 256:                         throw new Exception\Generic(
 257:                             "Unclosed \\" . $char . " character class"
 258:                         );
 259:                     }
 260:                     //Otherwise emit the unicode char class token
 261:                     else {
 262:                         $this->_emitToken(
 263:                             Token::TYPE_UNICODE_CHAR_CLASS,
 264:                             $char,
 265:                             $nextChar . $nextChars
 266:                         );
 267:                     }
 268:                 }
 269:             }
 270:             //If escaped and it's the hexadecimal character identifier "x"
 271:             elseif ($this->_escaped && $char === "x") {
 272:                 $nextChar = $this->_consume();
 273:                 $tokenSubject = "";
 274:                 if ($nextChar === "{") {
 275:                     $nextChars = $this->_consumeUntil("}", true);
 276:                     if ($nextChars === null) {
 277:                         throw new Exception\Generic(
 278:                             "Unclosed brace in hex char"
 279:                         );
 280:                     }
 281:                     $tokenSubject = trim($nextChars, "}");
 282:                 } elseif ($nextChar !== null) {
 283:                     $this->_unconsume();
 284:                     //Find following hexadecimal digits
 285:                     for ($i = 0; $i < 2; $i++) {
 286:                         $nextChar = $this->_consume();
 287:                         if ($nextChar !== null &&
 288:                             Rules::validateHexString($nextChar)) {
 289:                             $tokenSubject .= $nextChar;
 290:                         } else {
 291:                             $nextChar !== null && $this->_unconsume();
 292:                             break;
 293:                         }
 294:                     }
 295:                 }
 296:                 //Emit the hexadecimal character token
 297:                 $this->_emitToken(
 298:                     Token::TYPE_HEX_CHAR,
 299:                     $char,
 300:                     $tokenSubject
 301:                 );
 302:             }
 303:             //If escaped and it's the control character identifier "c"
 304:             elseif ($this->_escaped && $char === "c") {
 305:                 //Take the next character
 306:                 $nextChar = $this->_consumeIgnoreEscape();
 307:                 //If there are no characters left throw an exception
 308:                 if ($nextChar === null) {
 309:                     throw new Exception\Generic(
 310:                         "Character not specified for control character"
 311:                     );
 312:                 }
 313:                 //Otherwise emit the control character token
 314:                 $this->_emitToken(
 315:                     Token::TYPE_CONTROL_CHAR,
 316:                     $char,
 317:                     $nextChar
 318:                 );
 319:             }
 320:             //If not in char class and not escaped and it's a valid repetition
 321:             //identifier
 322:             elseif (!$this->_inCharClass && !$this->_escaped &&
 323:                     ($char === "*" || $char === "+" || $char === "?")) {
 324:                 //Emit a repetition token
 325:                 $this->_emitToken(
 326:                     Token::TYPE_REPETITION,
 327:                     $char
 328:                 );
 329:             }
 330:             //If not in char class and not escaped and it's an open curly brace
 331:             //and the following text identifies a repetition
 332:             elseif (!$this->_inCharClass && !$this->_escaped && $char === "{" &&
 333:                     $nextChars = $this->_consumeRegex("/^\d+(?:,\d*)?\}/")) {
 334:                 //Emit a repetition token
 335:                 $this->_emitToken(
 336:                     Token::TYPE_REPETITION,
 337:                     $char,
 338:                     rtrim($nextChars, "}")
 339:                 );
 340:             }
 341:             //If not in char class and not escaped and it's an open round bracket
 342:             elseif (!$this->_inCharClass && !$this->_escaped && $char === "(") {
 343:                 $this->_handleSubpattern();
 344:             }
 345:             //If not in char class and not escaped and it's a closed round bracket
 346:             elseif (!$this->_inCharClass && !$this->_escaped && $char === ")") {
 347:                 //Throw exception if there are no open subpatterns
 348:                 if (!$this->_openSubpatterns) {
 349:                     throw new Exception\Generic(
 350:                         "Unmatched parenthesis"
 351:                     );
 352:                 }
 353:                 //Emit a subpattern end token
 354:                 $this->_emitToken(
 355:                     Token::TYPE_SUBPATTERN_END,
 356:                     $char
 357:                 );
 358:                 $this->_openSubpatterns--;
 359:                 $this->_modifiersStack->pop();
 360:             }
 361:             //If not in char class and not escaped and it's an open square bracket
 362:             elseif (!$this->_inCharClass && !$this->_escaped && $char === "[") {
 363:                 //Emit a char class start token
 364:                 $this->_emitToken(
 365:                     Token::TYPE_CHAR_CLASS_START,
 366:                     $char
 367:                 );
 368:                 $this->_inCharClass = true;
 369:                 //Consume next char
 370:                 $char = $this->_consume();
 371:                 //If the character is a char class negation
 372:                 if ($char === "^") {
 373:                     //Emit the char class negate token
 374:                     $this->_emitToken(
 375:                         Token::TYPE_CHAR_CLASS_NEGATE,
 376:                         $char
 377:                     );
 378:                     $char = $this->_consume();
 379:                 }
 380:                 //If the first char in a char class is a closed square bracket
 381:                 if ($char === "]") {
 382:                     //Emit the bracket as char token
 383:                     $this->_emitToken(
 384:                         Token::TYPE_CHAR,
 385:                         $char
 386:                     );
 387:                 } else {
 388:                     $this->_unconsume();
 389:                 }
 390:             }
 391:             //If in char class and not escaped and it's a closed square bracket
 392:             elseif ($this->_inCharClass && !$this->_escaped && $char === "]") {
 393:                 //Emit a char class end token
 394:                 $this->_emitToken(
 395:                     Token::TYPE_CHAR_CLASS_END,
 396:                     $char
 397:                 );
 398:                 $this->_inCharClass = false;
 399:             }
 400:             //If in char class and it's not escaped and it's a dash, check if
 401:             //last token type is one of the allowed ones to make a char class
 402:             //range
 403:             elseif ($this->_inCharClass && !$this->_escaped && $char === "-" &&
 404:                     in_array(
 405:                         $this->_lastToken->getType(),
 406:                         $this->_allowedInCharClassRange
 407:                     )) {
 408:                 //Enable the after char class range mode
 409:                 $this->_afterCharClassRange = true;
 410:             }
 411:             //If in char class and it's a open square bracket and it's followed
 412:             //by the posix char class definition
 413:             elseif ($this->_inCharClass && !$this->_escaped && $char === "[" &&
 414:                     ($nextChars = $this->_consumeRegex("/^:\^?[a-z]+:\]/"))) {
 415:                 //Emit a posix char class token
 416:                 $subject = str_replace(array(":", "]"), "", $nextChars);
 417:                 $this->_emitToken(
 418:                     Token::TYPE_POSIX_CHAR_CLASS,
 419:                     $char . $nextChars,
 420:                     $subject
 421:                 );
 422:             }
 423:             //If not in char class and escaped and it's g or k
 424:             elseif (!$this->_inCharClass && $this->_escaped &&
 425:                     ($char === "g" || $char === "k")) {
 426:                 //It's a back reference. Check for the reference identifier
 427:                 if ($char === "g") {
 428:                     $testPattern = "(?|(\d+)|\{(-?\d+|\w+)\})";
 429:                 } else {
 430:                     $testPattern = "(<\w+>|'\w+'|\{\w+\})";
 431:                 }
 432:                 $nextChars = $this->_consumeRegex("/^$testPattern/", 1);
 433:                 if ($nextChars === null) {
 434:                     throw new Exception\Generic(
 435:                         "Invalid backreference"
 436:                     );
 437:                 }
 438:                 if ($char === "k") {
 439:                     $nextChars = substr($nextChars, 1, -1);
 440:                 }
 441:                 //Check reference validity
 442:                 if (!$this->_checkValidReference($nextChars)) {
 443:                     throw new Exception\Generic(
 444:                         "Reference to non-existent subpattern '$nextChars'"
 445:                     );
 446:                 }
 447:                 //Emit a backreference token
 448:                 $this->_emitToken(
 449:                     Token::TYPE_BACK_REFERENCE,
 450:                     $char,
 451:                     $nextChars
 452:                 );
 453:             }
 454:             //If escaped and it's a number
 455:             elseif ($this->_escaped && is_numeric($char)) {
 456:                 //Char class does not handle back references so if the character
 457:                 //is not octal process the character again without the escape
 458:                 if ($this->_inCharClass && $char > 7) {
 459:                     $this->_unconsume();
 460:                     $this->_escaped = false;
 461:                     continue;
 462:                 }
 463:                 //If the character is a 0 consume up to 2 octal digits,
 464:                 //otherwise consume all the following digits
 465:                 if ($char === "0" || $this->_inCharClass) {
 466:                     $testPattern = "^[0-7]{1,2}";
 467:                 } else {
 468:                     $testPattern = "^\d+";
 469:                 }
 470:                 //Consume following numbers
 471:                 $nextChars = $this->_consumeRegex("/^$testPattern/");
 472:                 if ($nextChars !== null) {
 473:                     $char .= $nextChars;
 474:                 }
 475:                 //If the first digit is 0 or its a valid octal number and there
 476:                 //are not enough back references
 477:                 $hasReference = $this->_checkValidReference($char);
 478:                 if ($char[0] === "0" || $this->_inCharClass ||
 479:                     (preg_match("/^[0-7]{2,3}$/", $char) && !$hasReference)) {
 480:                     $this->_emitToken(
 481:                         Token::TYPE_OCTAL_CHAR,
 482:                         $char
 483:                     );
 484:                 }
 485:                 //If there number corresponds to a subpattern
 486:                 elseif ($hasReference) {
 487:                     //Emit a backreference token
 488:                     $this->_emitToken(
 489:                         Token::TYPE_BACK_REFERENCE,
 490:                         "\\",
 491:                         $char
 492:                     );
 493:                 }
 494:                 //Otherwise the character does not match any subpattern so
 495:                 //throw an exception
 496:                 else {
 497:                     throw new Exception\Generic(
 498:                         "Reference to non-existent subpattern '$char'"
 499:                     );
 500:                 }
 501:             }
 502:             //If not escaped and it's the end delimiter
 503:             elseif (!$this->_escaped && $checkEndDelimiter &&
 504:                     $char === $endDelimiter) {
 505:                 //Throw an exception
 506:                 throw new Exception\InvalidDelimiter(
 507:                     "Unescaped end delimiter '$char' inside regex"
 508:                 );
 509:             }
 510:             //If it does not fall in any of the cases above
 511:             else {
 512:                 //If the character is not escaped and the "x" modifier is active
 513:                 if (!$this->_escaped &&
 514:                     strpos($this->_modifiersStack->top(), "x") !== false) {
 515:                     //If it is a "#"
 516:                     if ($char === "#") {
 517:                         //Emit a comment token
 518:                         $nextChars = $this->_consumeUntil("\n");
 519:                         if ($nextChars === null) {
 520:                             $nextChars = $this->_consumeRemaining();
 521:                         }
 522:                         $this->_emitToken(
 523:                             Token::TYPE_COMMENT,
 524:                             $char,
 525:                             $nextChars
 526:                         );
 527:                         continue;
 528:                     }
 529:                     //If it is a whitespace ignore it
 530:                     elseif (preg_match("/\s/", $char)) {
 531:                         continue;
 532:                     }
 533:                 }
 534:                 //Emit the character as a simple pattern token
 535:                 $this->_emitToken(
 536:                     Token::TYPE_CHAR,
 537:                     $char
 538:                 );
 539:             }
 540: 
 541:             //Reset the escaped state
 542:             $this->_escaped  = false;
 543:         }
 544: 
 545:         //If the escaped state is already active it means that no end delimiter
 546:         //has been found, so an exception must be thrown
 547:         if ($this->_escaped) {
 548:             throw new Exception\InvalidDelimiter(
 549:                 "End delimiter '$endDelimiter' not found"
 550:             );
 551:         }
 552: 
 553:         //Throw exception if there are unclosed subpatterns
 554:         if ($this->_openSubpatterns) {
 555:             throw new Exception\Generic(
 556:                 "The regex contains unclosed subpatterns"
 557:             );
 558:         }
 559: 
 560:         //Throw exception if there are unclosed char classes
 561:         if ($this->_inCharClass) {
 562:             throw new Exception\Generic(
 563:                 "The regex contains unclosed character classes"
 564:             );
 565:         }
 566: 
 567:         //Emit the end delimiter token
 568:         $this->_emitToken(
 569:             Token::TYPE_REGEX_END_DELIMITER,
 570:             $endDelimiter
 571:         );
 572: 
 573:         //If regex modifiers were specified emit the token
 574:         if ($rModifiers) {
 575:             $this->_emitToken(
 576:                 Token::TYPE_REGEX_MODIFIERS,
 577:                 $rModifiers
 578:             );
 579:         }
 580:     }

 582:     /**
 583:      * Handles every case that can happen after a open round bracket
 584:      * 
 585:      * @return void
 586:      * 
 587:      * @throws \REBuilder\Exception\Generic
 588:      */
 589:     protected function _handleSubpattern ()
 590:     {
 591:         //Don't emit the tokens immediately but store them in an array, in this
 592:         //way they can be handled before emitting
 593:         $tokens = array();
 594:         $tokens[] = array(
 595:             Token::TYPE_SUBPATTERN_START,
 596:             "("
 597:         );
 598:         $this->_applyModifiers("");
 599:         $this->_openSubpatterns++;
 600:         $nextChar = $this->_consume();
 601:         //Check if the next character is a question mark that identifies
 602:         //group options
 603:         if ($nextChar === "?") {
 604:             //Check if the following characters represent subpattern
 605:             //modifiers and/or non capturing flag
 606:             if ($nextChars = $this->_consumeRegex("/^[a-z\-]*:/i")) {
 607:                 //Store a non capturing subpattern token
 608:                 $groupModifiers = rtrim($nextChars, ":");
 609:                 if ($groupModifiers) {
 610:                     $this->_applyModifiers($groupModifiers);
 611:                 }
 612:                 $tokens[] = array(
 613:                     Token::TYPE_SUBPATTERN_NON_CAPTURING,
 614:                     ":",
 615:                     $groupModifiers
 616:                 );
 617:             }
 618:             //Check if the following character is a pipe
 619:             elseif ($nextChar = $this->_consumeIfEquals("|")) {
 620:                 //Store a subpattern group matches token
 621:                 $tokens[] = array(
 622:                     Token::TYPE_SUBPATTERN_GROUP_MATCHES,
 623:                     $nextChar
 624:                 );
 625:             }
 626:             //Check if the following character represents the once only
 627:             //subpattern identifier
 628:             elseif ($nextChar = $this->_consumeIfEquals(">")) {
 629:                 //Store a subpattern once only token
 630:                 $tokens[] = array(
 631:                     Token::TYPE_SUBPATTERN_ONCE_ONLY,
 632:                     $nextChar
 633:                 );
 634:             }
 635:             //Check if the following character represents the conditional
 636:             //subpattern identifier
 637:             elseif ($nextChar = $this->_consumeIfEquals("(")) {
 638:                 //Remove current tokens
 639:                 $tokens = array();
 640:                 //Store a conditional subpattern token
 641:                 $tokens[] = array(
 642:                     Token::TYPE_CONDITIONAL_SUBPATTERN,
 643:                     "(?"
 644:                 );
 645:                 //Unconsume the last character so that it will be processed
 646:                 //again
 647:                 $this->_unconsume();
 648:             }
 649:             //Check if the following characters represent subpattern
 650:             //name
 651:             elseif ($nextChars = $this->_consumeRegex(
 652:                         "/^(?|P?<(\w+)>|'(\w+)')/", 1
 653:                     )) {
 654:                 $this->_matches[] = $nextChars;
 655:                 //Store a subpattern name token
 656:                 $tokens[] = array(
 657:                     Token::TYPE_SUBPATTERN_NAME,
 658:                     "P",
 659:                     $nextChars
 660:                 );
 661:             }
 662:             //Check if the following characters represent a recursive pattern
 663:             elseif ($nextChars = $this->_consumeRegex("/^(R|\-?\d+)\)/", 1)) {
 664:                 //Check reference validity
 665:                 if ($nextChars === "R") {
 666:                     if (!count($this->_matches)) {
 667:                         throw new Exception\Generic(
 668:                             "Reference to non-existent subpattern '$nextChars'"
 669:                         );
 670:                     }
 671:                 } elseif (!$this->_checkValidReference($nextChars)) {
 672:                     throw new Exception\Generic(
 673:                         "Reference to non-existent subpattern '$nextChars'"
 674:                     );
 675:                 }
 676:                 //Remove current tokens
 677:                 $tokens = array();
 678:                 $this->_openSubpatterns--;
 679:                 //Store a recursive pattern token
 680:                 $tokens[] = array(
 681:                     Token::TYPE_RECURSIVE_PATTERN,
 682:                     "(?$nextChars)",
 683:                     $nextChars
